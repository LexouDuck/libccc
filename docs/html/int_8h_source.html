493#define DEFINEFUNC_Int(X, FUNCTYPE) \
494 _Generic((X), \
495 t_s16: FUNCTYPE##_FromInt, \
496 t_s32: FUNCTYPE##_FromInt, \
497 t_s64: FUNCTYPE##_FromInt, \
498 t_s128: FUNCTYPE##_FromInt, \
499 t_sint: FUNCTYPE##_FromInt, \
500 t_q16: FUNCTYPE##_FromFixed, \
501 t_q32: FUNCTYPE##_FromFixed, \
502 t_q64: FUNCTYPE##_FromFixed, \
503 t_q128: FUNCTYPE##_FromFixed, \
504 t_fixed: FUNCTYPE##_FromFixed, \
505 t_f32: FUNCTYPE##_FromFloat, \
506 t_f64: FUNCTYPE##_FromFloat, \
507 t_f80: FUNCTYPE##_FromFloat, \
508 t_f128: FUNCTYPE##_FromFloat, \
509 t_float: FUNCTYPE##_FromFloat, \
510 )(X)
511
512#define Int(X) DEFINEFUNC_Int(X, Int)
513#define S16(X) DEFINEFUNC_Int(X, S16)
514#define S32(X) DEFINEFUNC_Int(X, S32)
515#define S64(X) DEFINEFUNC_Int(X, S64)
516#define S128(X) DEFINEFUNC_Int(X, S128)
