661#define INCBIN(NAME, FILEPATH) \
662/*extern t_u8 const NAME[]; */ \
663/*extern t_u8 const NAME##_end[]; */ \
664/*extern int const NAME##_size[]; */ \
665__asm__ \
666( \
667 "\n"INCBIN_SECTION \
668 "\n" \
669 "\n"INCBIN_GLOBAL" " \
670 INCBIN_MANGLE(#NAME) \
671 "\n"INCBIN_MANGLE(#NAME)":" \
672 "\n\t.incbin \""FILEPATH"\"" \
673 "\n" \
674 "\n"INCBIN_GLOBAL" " \
675 INCBIN_MANGLE(#NAME)"_end" \
676 "\n"INCBIN_MANGLE(#NAME)"_end"":" \
677 "\n\t"INCBIN_BYTE" 0" \
678 "\n" \
679 "\n"INCBIN_GLOBAL" " \
680 INCBIN_MANGLE(#NAME)"_size" \
681 "\n"INCBIN_MANGLE(#NAME)"_size"":" \
682 "\n\t"INCBIN_SIZE \
683 " ( "INCBIN_MANGLE(#NAME)"_end" \
684 " - "INCBIN_MANGLE(#NAME)" )" \
685 "\n" \
686 "\n"INCBIN_PREVIOUS \
687 "\n" \
688);
