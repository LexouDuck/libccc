350#define DEFINEFUNC_Fixed(X, FUNCTYPE) \
351 _Generic((X), \
352 t_s16: FUNCTYPE##_FromInt, \
353 t_s32: FUNCTYPE##_FromInt, \
354 t_s64: FUNCTYPE##_FromInt, \
355 t_s128: FUNCTYPE##_FromInt, \
356 t_sint: FUNCTYPE##_FromInt, \
357 t_q16: FUNCTYPE##_FromFixed, \
358 t_q32: FUNCTYPE##_FromFixed, \
359 t_q64: FUNCTYPE##_FromFixed, \
360 t_q128: FUNCTYPE##_FromFixed, \
361 t_fixed: FUNCTYPE##_FromFixed, \
362 t_f32: FUNCTYPE##_FromFloat, \
363 t_f64: FUNCTYPE##_FromFloat, \
364 t_f80: FUNCTYPE##_FromFloat, \
365 t_f128: FUNCTYPE##_FromFloat, \
366 t_float: FUNCTYPE##_FromFloat, \
367 )(X)
368#define Fixed(X) DEFINEFUNC_Fixed(X, Fixed)
369#define Q16(X) DEFINEFUNC_Fixed(X, Q16)
370#define Q32(X) DEFINEFUNC_Fixed(X, Q32)
