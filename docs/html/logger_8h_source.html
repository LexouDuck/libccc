157#define LOGONE_FUNCTION_CONTENT(SKIP, ERRORINT, ERRORMSG, PREFIX, PREFIX_COLOR) \
158 e_cccerror result = ERROR_NONE; \
159 va_list args; \
160 if (SKIP) return (result); \
161 va_start(args, format_str); \
162 result = Log_VA(logger, \
163 ERRORINT, \
164 PREFIX, \
165 PREFIX_COLOR, \
166 ERRORMSG, \
167 format_str, args); \
168 va_end(args); \
169 return (result); \
170
171#define LOGALL_FUNCTION_CONTENT(SKIP, ERRORINT, ERRORMSG, PREFIX, PREFIX_COLOR) \
172 e_cccerror result = ERROR_NONE; \
173 va_list args; \
174 for (t_u32 i = 0; loggers[i]; ++i) \
175 { \
176 s_logger const* logger = loggers[i]; \
177 if (SKIP) continue; \
178 va_start(args, format_str); \
179 result = Log_VA(logger, \
180 ERRORINT, \
181 PREFIX, \
182 PREFIX_COLOR, \
183 ERRORMSG, \
184 format_str, args); \
185 va_end(args); \
186 } \
187 return (result); \
