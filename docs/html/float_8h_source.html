451#define DEFINEFUNC_Float(X, FUNCTYPE) \
452 _Generic((X), \
453 t_s16: FUNCTYPE##_FromInt, \
454 t_s32: FUNCTYPE##_FromInt, \
455 t_s64: FUNCTYPE##_FromInt, \
456 t_s128: FUNCTYPE##_FromInt, \
457 t_sint: FUNCTYPE##_FromInt, \
458 t_q16: FUNCTYPE##_FromFixed, \
459 t_q32: FUNCTYPE##_FromFixed, \
460 t_q64: FUNCTYPE##_FromFixed, \
461 t_q128: FUNCTYPE##_FromFixed, \
462 t_fixed: FUNCTYPE##_FromFixed, \
463 t_f32: FUNCTYPE##_FromFloat, \
464 t_f64: FUNCTYPE##_FromFloat, \
465 t_f80: FUNCTYPE##_FromFloat, \
466 t_f128: FUNCTYPE##_FromFloat, \
467 t_float: FUNCTYPE##_FromFloat, \
468 )(X)
469

471#define F32(X) DEFINEFUNC_Float(X, F32)
472#define F64(X) DEFINEFUNC_Float(X, F64)
