468#define DEFINEFUNC_Float(X, FUNCTYPE) \
469 _Generic((X), \
470 t_s16: FUNCTYPE##_FromInt, \
471 t_s32: FUNCTYPE##_FromInt, \
472 t_s64: FUNCTYPE##_FromInt, \
473 t_s128: FUNCTYPE##_FromInt, \
474 t_sint: FUNCTYPE##_FromInt, \
475 t_q16: FUNCTYPE##_FromFixed, \
476 t_q32: FUNCTYPE##_FromFixed, \
477 t_q64: FUNCTYPE##_FromFixed, \
478 t_q128: FUNCTYPE##_FromFixed, \
479 t_fixed: FUNCTYPE##_FromFixed, \
480 t_f32: FUNCTYPE##_FromFloat, \
481 t_f64: FUNCTYPE##_FromFloat, \
482 t_f80: FUNCTYPE##_FromFloat, \
483 t_f128: FUNCTYPE##_FromFloat, \
484 t_float: FUNCTYPE##_FromFloat, \
485 )(X)

488#define F32(X) DEFINEFUNC_Float(X, F32)
