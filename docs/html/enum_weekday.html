  • Standard library equivalent: (C) https://en.cppreference.com/w/c/chrono/clock_t

This enum lists each day of the week, Sunday counting as zero

#include <time.h>

enum weekday
{
    WEEKDAY_SUNDAY,
    WEEKDAY_MONDAY,
    WEEKDAY_TUESDAY,
    WEEKDAY_WEDNESDAY,
    WEEKDAY_THURSDAY,
    WEEKDAY_FRIDAY,
    WEEKDAY_SATURDAY,
    ENUMLENGTH_WEEKDAY,
};
